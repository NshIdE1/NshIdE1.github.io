        input += 16;
        output += 16;
        length -= 16;

                   unsigned char *output)

    {
        while (length > 0)
        {
            for (i = 0; i < 16; i++)
                output[i] = (unsigned char)(input[i] ^ iv[i]);

            output += 16;
            length -= 16;
        }
    }
}
