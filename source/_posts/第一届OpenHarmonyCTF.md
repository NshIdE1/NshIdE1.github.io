def KSA(key):
    S = list(range(256))
    j = 0
    for i in range(256):
        j = (j + S[j] + key[j % len(key)]) % 256
        S[i], S[j] = S[j], S[i]
    return S

def PRGA(S):
    i = j = 0
    while True:
        i = (i + 1) % 256
        j = (j + S[i]) % 256
        S[i], S[j] = S[j], S[i]
        K = S[(S[i] + S[j]) % 256]
        yield K

def rc4_decrypt(key: bytes, cipher_bytes: bytes) -> bytes:
    S = KSA(list(key))
    keystream = PRGA(S)
    decrypted = bytes((c - next(keystream)) % 256 for c in cipher_bytes)
    return decrypted
